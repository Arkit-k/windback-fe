## Using the API directly

```go
package main

import (
    "fmt"
    "io"
    "net/http"
    "os"
    "strings"
)

func main() {
    apiKey := os.Getenv("WINDBACK_SECRET_KEY") // sk_...
    base := "https://api.windbackai.com/api/v1"

    // Get stats
    req, _ := http.NewRequest("GET", base+"/projects/my-app/stats", nil)
    req.Header.Set("X-API-Key", apiKey)
    resp, _ := http.DefaultClient.Do(req)
    body, _ := io.ReadAll(resp.Body)
    resp.Body.Close()
    fmt.Println(string(body))

    // Track an event
    req, _ = http.NewRequest("POST", base+"/track", strings.NewReader(
        `{"customer_email":"user@example.com","event":"login"}`,
    ))
    req.Header.Set("X-API-Key", apiKey)
    req.Header.Set("Content-Type", "application/json")
    http.DefaultClient.Do(req)
}
```
