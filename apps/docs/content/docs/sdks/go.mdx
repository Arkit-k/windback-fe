# Go SDK

<Callout type="warn">
The Go SDK is not yet published. Use the [REST API](/docs/api-reference) directly for now.
</Callout>

## Using the API directly

```go
package main

import (
    "fmt"
    "io"
    "net/http"
    "os"
)

func main() {
    apiKey := os.Getenv("WINDBACK_SECRET_KEY") // sk_...
    base := "https://api.windbackai.com/api/v1"

    // Get stats
    req, _ := http.NewRequest("GET", base+"/projects/my-app/stats", nil)
    req.Header.Set("X-API-Key", apiKey)

    resp, err := http.DefaultClient.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := io.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
